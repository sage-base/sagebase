案2: D3.js + Mermaid

D3.jsで俯瞰し、 Mermaidで詳細を読む

インタラクティブなフォースレイアウトで全体像を掴み、
ER図でテーブル定義の詳細を確認できる二段構成です。

{{- /* Section 01 — D3.js フォースレイアウト */ -}}
01

全体俯瞰 — インタラクティブ関連図

ノードの形状でエンティティの種別を区別。ホバーで関連をハイライト、クリックで詳細を表示します。

{{- /* 形状の凡例 */ -}}
議会系
発言系
政治家系
議案系
{{- /* Section 02 — Mermaid.js ER図（詳細テーブル定義） */ -}}
02

テーブル定義 — ER図（詳細）

BigQueryスキーマに基づく各テーブルのカラム定義とリレーションを表示します。

{{- /* 議会議論系テーブル */ -}}

議会議論データ

erDiagram
    governing_bodies["開催主体 (governing_bodies)"] {
        INT64 id PK "開催主体ID"
        STRING name "名称"
        STRING organization_code "自治体コード"
        STRING organization_type "組織種別"
        STRING prefecture "都道府県"
    }
    conferences["議会 (conferences)"] {
        INT64 id PK "議会ID"
        STRING name "議会名"
        INT64 governing_body_id FK "開催主体ID"
        STRING term "任期"
        INT64 election_id FK "選挙ID"
    }
    meetings["会議 (meetings)"] {
        INT64 id PK "会議ID"
        INT64 conference_id FK "議会ID"
        DATE date "開催日"
        STRING name "会議名"
        STRING url "URL"
    }
    minutes["議事録 (minutes)"] {
        INT64 id PK "議事録ID"
        INT64 meeting_id FK "会議ID"
        STRING url "URL"
        TIMESTAMP processed_at "処理日時"
    }
    conversations["発言 (conversations)"] {
        INT64 id PK "発言ID"
        INT64 minutes_id FK "議事録ID"
        INT64 speaker_id FK "発言者ID"
        STRING comment "発言内容"
        INT64 sequence_number "発言順序"
    }
    speakers["発言者 (speakers)"] {
        INT64 id PK "発言者ID"
        STRING name "氏名"
        STRING political_party_name "政党名"
        STRING position "役職"
        INT64 politician_id FK "政治家ID"
        NUMERIC matching_confidence "照合信頼度"
        BOOL is_manually_verified "手動検証済み"
    }

    governing_bodies ||--o{ conferences : "has many"
    conferences ||--o{ meetings : "has many"
    meetings ||--o| minutes : "has one"
    minutes ||--o{ conversations : "has many"
    conversations }o--|| speakers : "belongs to"

erDiagram
    politicians["政治家 (politicians)"] {

    parliamentary_groups["会派 (parliamentary_groups)"] {
        INT64 id PK "会派ID"

    parliamentary_group_parties["会派-政党 (中間)"] {

    }
    parliamentary_group_memberships["会派所属履歴"] {
        INT64 id PK "所属ID"

        DATE end_date "終了日"
        STRING role "役割"
    }
    conference_members["議会構成員 (conference_members)"] {
        INT64 id PK "所属ID"

        STRING role "役割"

    election_members["選挙結果 (election_members)"] {
        INT64 id PK "選挙結果ID"
        INT64 election_id FK "選挙ID"
        INT64 politician_id FK "政治家ID"
        STRING result "結果"
        INT64 votes "得票数"
        INT64 rank "順位"
    }

    politicians ||--o{ parliamentary_group_memberships : "会派所属"
    parliamentary_group_memberships }o--|| parliamentary_groups : "所属先"
    parliamentary_groups }o--o{ political_parties : "構成政党"
    parliamentary_group_parties }o--|| parliamentary_groups : ""
    parliamentary_group_parties }o--|| political_parties : ""
    politicians ||--o{ election_members : "出馬"
    elections ||--o{ election_members : "候補者"
    politicians ||--o{ conference_members : "議会参加"
